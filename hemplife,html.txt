        // Age Verification Gate
        function checkAgeGate() {
            const isVerified = sessionStorage.getItem('ageVerified');
            if (isVerified === 'true') {
                document.getElementById('age-modal').style.display = 'none';
                document.body.style.overflow = 'auto';
            } else {
                document.getElementById('age-modal').style.display = 'flex';
                document.body.style.overflow = 'hidden';
            }
        }

            sessionStorage.setItem('ageVerified', 'true');

        // Routing System
        function navigate(pageId) {
            document.querySelectorAll('.page-view').forEach(page => {
                page.classList.remove('active');
            });
            
            const targetPage = document.getElementById('page-' + pageId);
            if (targetPage) {
                targetPage.classList.add('active');
            }
            
            document.querySelectorAll('.nav-link').forEach(link => {
                link.classList.remove('active');
                if (link.getAttribute('data-target') === pageId) {
                    link.classList.add('active');
                }
            });

            document.getElementById('mobile-menu').classList.remove('open');
            window.scrollTo({top: 0, behavior: 'smooth'});
        }

        // Mobile Menu Toggle
        function toggleMenu() {
            document.getElementById('mobile-menu').classList.toggle('open');
        }

        // Referral Tracking System
        function handleReferral() {
            const urlParams = new URLSearchParams(window.location.search);
            const urlRef = urlParams.get('ref');
            const urlName = urlParams.get('name');
            
            if (urlRef && urlName) {
                sessionStorage.setItem('refCode', urlRef);
                sessionStorage.setItem('refName', urlName);
                
                const newUrl = window.location.protocol + "//" + window.location.host + window.location.pathname;
                window.history.pushState({path:newUrl}, '', newUrl);
            }

            const sessionRef = sessionStorage.getItem('refCode');
            const sessionName = sessionStorage.getItem('refName');

            if (sessionRef && sessionName) {
                const banner = document.getElementById('referral-banner');
                banner.textContent = `Partner Referral Active: ${sessionName.replace('+', ' ')} / Code: ${sessionRef}`;
                banner.style.display = 'block';

                const codeInput = document.getElementById('form-ref-code');
                const nameInput = document.getElementById('form-ref-name');
                if (codeInput) codeInput.value = sessionRef;
                if (nameInput) nameInput.value = sessionName.replace('+', ' ');
            }
        }

        // Form Submission (Demo)
        function submitForm(e) {
            e.preventDefault();
            alert("Application received. Our compliance team will review and respond shortly.");
            document.getElementById('wholesale-form').reset();
            handleReferral(); 
        }

        // Initialize on Load
        document.addEventListener('DOMContentLoaded', () => {
            checkAgeGate();
            handleReferral();
        });
